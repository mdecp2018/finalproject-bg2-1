git基本指令

git add (註冊檔案或目錄到索引)

git commit (提交添加到索引的檔案)

git status (顯示修改檔案清單)

git log (顯示提交記錄)

git rm <file> (刪除檔案)

git show <commit> (查看提交的詳細記錄)

git clean (從工作目錄刪除非管理對象的檔案)

git checkout -- <file> (還原在工作目錄已更改的檔案)

git add -u (只註冊已提交過的檔案到索引)

git branch (列出所有 branch)

git pull (更新遠端回來)

git push (送檔案到遠端)

git merge (合併檔案)

from browser import document as doc
import turtle
turtle.set_defaults(
    turtle_canvas_wrapper = doc['brython_div']
)
painter = turtle.Turtle()

painter.pencolor("pink")

for i in range(50):
    painter.forward(80)
    painter.left(123) # Let's go counterclockwise this time 
    
painter.pencolor("green")
for i in range(50):
    painter.forward(160)
    painter.left(123) # Let's go counterclockwise this time
    
painter.pencolor("black")
for i in range(50):
    painter.forward(240)
    painter.left(123) # Let's go counterclockwise this time     

turtle.done()

# Turtle6 繪圖
from browser import document as doc
import turtle
import math
turtle.set_defaults(
    turtle_canvas_wrapper = doc['brython_div']
)

t = turtle.Turtle("turtle")
t.speed(10)

colors = ['red', 'orange', 'yellow', 'green', 'blue']
for x in range(250):
    t.pencolor(colors[x % 5])
    t.width(x/10 + 1)
    t.forward(x)
    t.left(100)
turtle.done()
